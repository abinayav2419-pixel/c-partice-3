#include <stdio.h>
#include <stdlib.h>

int main() {
    int n; scanf("%d",&n);
    int arr[n], closest=0;
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
        if(i==0) closest=arr[i];
        else if(abs(arr[i])<abs(closest)) closest=arr[i];
        else if(abs(arr[i])==abs(closest) && arr[i]>0) closest=arr[i];
    }
    printf("%d", closest);
    return 0;
}
