#include <stdio.h>

int main() {
    int R, C;
    scanf("%d %d", &R, &C);
    int mat[R*C];
    
    for(int i = 0; i < R*C; i++)
        scanf("%d", &mat[i]);

    int maxCount = 0, freqVal = mat[0];
    
    for(int i = 0; i < R*C; i++) {
        int count = 0;
        for(int j = 0; j < R*C; j++) {
            if(mat[i] == mat[j]) count++;
        }
        if(count > maxCount) {
            maxCount = count;
            freqVal = mat[i];
        }
    }

    printf("%d\n", freqVal);
    return 0;
}
