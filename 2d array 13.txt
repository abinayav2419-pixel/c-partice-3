#include <stdio.h>

int main() {
    int R, C;
    scanf("%d %d", &R, &C);
    int mat[R][C];
    
    for(int i = 0; i < R; i++)
        for(int j = 0; j < C; j++)
            scanf("%d", &mat[i][j]);

    int count = 0;
    for(int i = 0; i < R; i++) {
        int palin = 1;
        for(int j = 0; j < C/2; j++) {
            if(mat[i][j] != mat[i][C-1-j]) {
                palin = 0;
                break;
            }
        }
        if(palin) count++;
    }

    printf("%d\n", count);
    return 0;
}
