#include <stdio.h>

int main() {
    int R, C;
    scanf("%d %d", &R, &C);
    int mat[R][C];
    
    for(int i = 0; i < R; i++)
        for(int j = 0; j < C; j++)
            scanf("%d", &mat[i][j]);

    int maxSum = -2147483648, colIndex = 0;

    for(int j = 0; j < C; j++) {
        int sum = 0;
        for(int i = 0; i < R; i++)
            sum += mat[i][j];
        if(sum > maxSum) {
            maxSum = sum;
            colIndex = j;
        }
    }

    printf("%d\n", colIndex);
    return 0;
}
