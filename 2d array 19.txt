#include <stdio.h>

int main() {
    int R, C;
    scanf("%d %d", &R, &C);
    int matrix[R][C];
    for(int i=0;i<R;i++)
        for(int j=0;j<C;j++)
            scanf("%d", &matrix[i][j]);

    int freq[10000] = {0}; // adjust size if needed
    for(int i=0;i<R;i++)
        for(int j=0;j<C;j++)
            freq[matrix[i][j]]++;

    for(int i=0;i<R;i++){
        for(int j=0;j<C;j++){
            if(freq[matrix[i][j]]==1){
                printf("%d\n", matrix[i][j]);
                return 0;
            }
        }
    }
    return 0;
}
