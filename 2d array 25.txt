#include <stdio.h>

int main() {
    int R, C;
    scanf("%d %d", &R, &C);
    int matrix[R][C];
    for(int i=0;i<R;i++)
        for(int j=0;j<C;j++)
            scanf("%d", &matrix[i][j]);
    int count=0;
    for(int i=0;i<R;i++){
        int mn=matrix[i][0], mx=matrix[i][0];
        for(int j=1;j<C;j++){
            if(matrix[i][j]<mn) mn=matrix[i][j];
            if(matrix[i][j]>mx) mx=matrix[i][j];
        }
        if(mx-mn<=1000) count++;
    }
    printf("%d\n", count);
    return 0;
}
