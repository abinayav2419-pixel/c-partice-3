#include <stdio.h>

int main() {
    int r, c, k = 0;

    scanf("%d %d", &r, &c);

    int a[r][c];
    int b[r * c];

    for(int i = 0; i < r; i++) {
        for(int j = 0; j < c; j++) {
            scanf("%d", &a[i][j]);
        }
    }

    for(int i = 0; i < r; i++) {
        for(int j = 0; j < c; j++) {
            b[k++] = a[i][j];
        }
    }

    
    for(int i = 0; i < r * c; i++) {
        for(int j = i + 1; j < r * c; j++) {
            if(b[i] == b[j]) {
                printf("%d", b[i]);
                return 0;
            }
        }
    }

    printf("No repeating element");
    return 0;
}
