#include <stdio.h>
#include <math.h>
int isArmstrong(int num) {
    int originalNum = num, remainder, n = 0;
    double sum = 0.0;
    while (originalNum != 0) {
        originalNum /= 10;
        n++;
    }

    originalNum = num;
    while (originalNum != 0) {
        remainder = originalNum % 10;
        sum += pow(remainder, n);
        originalNum /= 10;
    }

    return ((int)sum == num);
}

int main() {
    int num;
    scanf("%d", &num);
    if (isArmstrong(num))
        printf("Armstrong number");
    else
        printf("Not Armstrong number");

    return 0;
}
