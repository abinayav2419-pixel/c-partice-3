 #include<stdio.h>
  int countdigit(int n){
      int count = 0;
      if(n==0)
      return 1;
      
        while(n!=0){
        n/=10;
        count++;
  }
      return count;
  }
  int main(){
      int N;
      scanf("%d",&N);
      printf("%d",countdigit(N));
      return 0;
  }
