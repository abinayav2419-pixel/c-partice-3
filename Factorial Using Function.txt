#include <stdio.h>

   int factorial(int n) {
    if (n == 0 || n == 1)
        return 1; 
    return n * factorial(n - 1); 
}

int main() {
    int N;
    scanf("%d", &N);

    if (N < 0) {
        printf("Enter positive value");
    } else {
        printf("%d\n", factorial(N));
    }

    return 0;
}
