#include <stdio.h>
#include <ctype.h>
int main() {
    char str[100];
    int i, consonants = 0;
    
    fgets(str, sizeof(str), stdin);

    for(i = 0; str[i] != '\0'; i++) {
        char ch = tolower(str[i]);

        if(isalpha(ch)) {
            if(ch!='a' && ch!='e' && ch!='i' && ch!='o' && ch!='u') {
                consonants++;
            }
        }
    }
    printf(" %d\n", consonants);
    return 0;
}
