#include <stdio.h>
#include <string.h>

int main() {
    char word[100];
    int length, i;

    scanf("%s", word);  
    length = strlen(word);

    for(i = 0; i < length / 2; i++) {
        if(word[i] != word[length - i - 1]) {
            printf("Not a palindrome.\n", word);
            return 0; 
        }
    }

    printf("palindrome.\n", word);
    return 0;
}
