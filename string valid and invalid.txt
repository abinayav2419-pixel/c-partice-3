#include <stdio.h>

int main() {
    char str[100];
    int i;
    int hasUppercase = 0;  
    scanf("%s", str);

    for(i = 0; str[i] != '\0'; i++) {
        if(str[i] >= 'A' && str[i] <= 'Z') {
            hasUppercase = 1;  
            break;            
        }
    }

    if(hasUppercase) {
        printf("Valid\n");
    } else {
        printf("Invalid\n");
    }

    return 0;
}
